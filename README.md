CREATE TABLE subjects (  
    id INT AUTO_INCREMENT PRIMARY KEY,  
    name VARCHAR(100) NOT NULL,  
    units INT NOT NULL,  
    gwa FLOAT NOT NULL,  
    included BOOLEAN DEFAULT TRUE,  
    tag VARCHAR(50) DEFAULT ''
